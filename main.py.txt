import os
import base64
from datetime import datetime, timedelta
from pathlib import Path
from google.auth.transport.requests import Request
from google.oauth2.credentials import Credentials
from google_auth_oauthlib.flow import InstalledAppFlow
from googleapiclient.discovery import build

# הגדרת הרשאות הגישה למייל וליומן כפי שנדרש במטלה
SCOPES = [
    "https://www.googleapis.com/auth/gmail.modify",
    "https://www.googleapis.com/auth/calendar"
]

CREDENTIALS_FILE = "credentials.json"
TOKEN_FILE = "token.json"

def get_credentials():
    creds = None
    if Path(TOKEN_FILE).exists():
        creds = Credentials.from_authorized_user_file(TOKEN_FILE, SCOPES)
    
    if not creds or not creds.valid:
        if creds and creds.expired and creds.refresh_token:
            print("מחדש את תוקף החיבור...")
            creds.refresh(Request())
        else:
            if not Path(CREDENTIALS_FILE).exists():
                print("\n❌ שגיאה: קובץ credentials.json עדיין לא נמצא בתיקייה שלך!")
                print("אל דאגה, מיד נטפל בזה ביחד.")
                return None
            
            print("פותח חלון דפדפן לאישור התחברות לגוגל...")
            flow = InstalledAppFlow.from_client_secrets_file(CREDENTIALS_FILE, SCOPES)
            creds = flow.run_local_server(port=0)
            
        Path(TOKEN_FILE).write_text(creds.to_json(), encoding="utf-8")
    return creds

def main():
    print("🚀 מפעיל את סוכן הסטודנטית האוטונומי...")
    creds = get_credentials()
    if not creds:
        return
        
    print("✅ החיבור לגוגל הצליח!")
    # כאן בהמשך נשבץ את לוגיקת קריאת המיילים והרובוט
    
if __name__ == "__main__":
    main()
